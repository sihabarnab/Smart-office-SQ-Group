Route::get('/{id1}/{id2}', function($id1,$id2)
 {
    $dec = base64_decode($id2);
    $user_id = base64_decode($id1);
    $files = DB::table('pro_file')->where('user_id',$user_id)->get();

	foreach ($files as $file)
	{
		if (Hash::check($file->file_id, $dec))
		{
			$m_file = DB::table('pro_file')->where('file_id', $file->file_id)->first();
			if ($m_file)
			{
				return response()->download($m_file->myFile);
			} else {
				return "File not found .<br> Please, Talk To Developer.";
			}
		}
	}

    
});
